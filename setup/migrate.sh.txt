#!/bin/bash
SCRIPT=$(readlink -f $0)
SCRIPTPATH=`dirname $SCRIPT`
CMD="php $SCRIPTPATH/migrate.php"
ERRMSG="Usage: migrate up|down|-v <version>"
if [[ $1 ]]
then
  if [ "$1" != "-v" ]
  then
    $CMD $1
  else
    if [[ $2 ]]
    then
        $CMD $1 $2
    else
        echo $ERRMSG
    fi
  fi
else
  echo $ERRMSG
fi
